314 template <size_t columns, size_t rows>
315 void LedController<columns,rows>::spiTransfer(unsigned int segment, byte opcode, byte data) {
316  if (!initilized || segment >= conf.SegmentCount()) {
317  return;
318  }
319 
320  // Create an array with the data to shift out
321  unsigned int offset = conf.getColumn(segment) * 2;
322  unsigned int row = conf.getRow(segment);
323  unsigned int maxbytes = conf.SegmentCount() * 2;
324 
325  for (unsigned int j = 0; j < rows;j++){
326  for (unsigned int i = 0; i < columns*2; i++) {
327  spidata[j][i] = 0x00;
328  }
329  }
330 
331  // put our device data into the array
332  spidata[row][offset + 1] = opcode;
333  spidata[row][offset] = data;
334 
335  for(int r = 0; r < rows ;r++){
336 
337  //enable the line
338  auto cs = conf.virtual_multi_row ? conf.SPI_CS : conf.row_SPI_CS[r];
339  digitalWrite(cs, LOW);
340 
341  //init the spi transfer if hardware should be used
342  if (conf.useHardwareSpi) {
343  SPI.beginTransaction(SPISettings(conf.spiTransferSpeed, MSBFIRST, SPI_MODE0));
344  }
345 
346  // Now shift out the data
347  for (int i = maxbytes; i > 0; i--) {
348  if (conf.useHardwareSpi) {
349  SPI.transfer(spidata[r][i - 1]);
350  } else {
351  shiftOut(conf.SPI_MOSI, conf.SPI_CLK, MSBFIRST, spidata[r][i - 1]);
352  }
353  }
354 
355  //end the spi transfer if hardware should be used
356  if (conf.useHardwareSpi) {
357  SPI.endTransaction();
358  }
359 
360  // latch the data onto the display
361  digitalWrite(cs, HIGH);
362  }
363 }
